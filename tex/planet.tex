	\varepsilon = \frac{L}{4\pi r^2}.

	\text{где}\;

\end{align*}

	T_E = \sqst[4]{\frac{\varepsilon \left( 1- a \right) }{4\sigma}},

	\text{где}\;

\end{align*}
